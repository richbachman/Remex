14 Forms

The framework includes default form styling for various input, select, textarea, radio, checkbox, and label fields. The labes are stacked above each form field. This is a very simple/basic form setup so feel free to customize.

Example

<form action="" method="post">
    <div class="w__4_5 m__auto">
        <label for="name">Name:</label>
        <input type="text" id="name" name="user_name">
    </div>
    <div class="w__4_5 m__auto">
        <label for="mail">E-mail:</label>
        <input type="email" id="mail" name="user_mail">
    </div>
    <div class="w__4_5 m__auto">
        <label for="disabled">Disabled</label>
        <input type="email" id="disabled" name="disabled" disabled>
    </div>
    <div class="w__4_5 m__auto">
        <label for="msg">Message:</label>
        <textarea id="msg" name="user_message"></textarea>
    </div>
    <div class="w__4_5 m__auto">
        <input type="radio" id="radio" name="radio"/>
        <label for="radio">Radio Input</label>
    </div>
    <div class="w__4_5 m__auto">
        <input type="radio" id="radio-disabled" name="radio-disabled" disabled/>
        <label for="radio-disabled">Disabled Radio Input</label>
    </div>
    <div class="w__4_5 m__auto">
        <input type="checkbox" id="checkbox" name="checkbox"/>
        <label for="checkbox">Checkbox Input</label>
    </div>
    <div class="w__4_5 m__auto m__bottom__md">
        <input type="checkbox" id="checkbox-disabled" name="checkbox-disabled" disabled/>
        <label for="checkbox-disabled">Disabled Checkbox Input</label>
    </div>
    <div class="w__4_5 m__auto">
        <label for="select">Select Option:</label>
        <select name="select" id="select">
            <option value="Test">Test</option>
            <option value="Test">Test</option>
            <option value="Test">Test</option>
            <option value="Test">Test</option>
        </select>
    </div>
</form>
